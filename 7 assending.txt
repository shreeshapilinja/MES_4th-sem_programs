;Write a program to arrange a series of 32 bit numbers in ascending order.
	AREA SEVENA, CODE, READONLY
	MOV R0,#4
	MOV R1,R0
UP MOV R2,#0X40000000
LOOP LDR R3,[R2],#4
	LDR R4,[R2]
	CMP R3,R4
	BLO DOWN
	STR R3,[R2],#-4
	STR R4,[R2],#4
DOWN
	SUB R0,R0,#1
	CMP R0,#0
	BNE LOOP
	MOV R0,R1
	SUB R1,R1,#1
	CMP R1,#0
	BNE UP
STOP B STOP
	END
