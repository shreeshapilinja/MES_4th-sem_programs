;Write a program to count the number of ones and zeros in two consecutive memory locations.
	AREA EIGHT, CODE, READONLY
	MOV R5,#2
	LDR R1,=ARRAY
UP LDR R2,[R1],#4
	MOV R0,#32
LOOP RORS R2,#1
	ADDCS R3,R3,#1
	ADDCC R4,R4,#1
	SUB R0,R0,#1
	CMP R0,#0
	BNE LOOP
	SUB R5,R5,#1
	CMP R5,#0
	BNE UP
STOP B STOP
ARRAY
	DCD 4,4
	ALIGN
	END
