// frc-1 to LED
// Required --> led card , arm 7 trainer kit ,power adapter,usb connector
#include<LPC214X.H>
unsigned int j=0;
void ext_interrupt(void)__irq
{
	IO0SET |= (0xFF<<16);
	for(j=0;j<=1000000;j++);
	IO0CLR |= (0xFF<<16);
	EXTINT = 0x02;
	VICVectAddr = 0x00000000;
}
void init_ext(void)
{
	IO0DIR |= (0xFF<<16);
	IO0CLR |= (0xFF<<16);
	PINSEL0 |= 0x20000000;
	EXTMODE = 0x02;
	EXTPOLAR = 0x02;
	VICVectAddr0 = (unsigned int)ext_interrupt;
	VICVectCntl0 = 0x000002F;
	VICIntEnable = 0x00008000;
}
int main(void)
{
	init_ext();
	while(1)
	{
	}
}
