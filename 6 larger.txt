	AREA SIXA, CODE, READONLY
	MOV R0,#6
	LDR R1,=ARRAY
	LDR R2,[R1]
LOOP ADD R1,R1,#4
	LDR R3,[R1]
	CMP R2,R3
	BHI DOWN
	MOV R2,R3
DOWN
	SUB R0,R0,#1
	CMP R0,#0
	BNE LOOP
ARRAY

	ALIGN

	END
